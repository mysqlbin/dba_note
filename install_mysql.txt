1. 创建一个帐号
groupadd mysql
useradd -g mysql -d /usr/local/mysql -s /sbin/nologin -M mysql
id mysql

2. 基本软件安装
解压目录
mkdir /opt/mysql
cd /opt/mysql
tar zxvf /vagrant/MySQL/mysql-5.7.19-linux-glibc2.12-x86_64.tar.gz
安装目录
cd /usr/local/
ln -s /opt/mysql/mysql-5.7.22-linux-glibc2.12-x86_64 mysql
ln -s /opt/mysql/mysql-5.7.19-linux-glibc2.12-x86_64 mysql


ln -s /opt/mysql/xenon-tokudb-mysql-5.7.19 mysql


chown -R mysql:mysql mysql/

chown mysql:mysql /data/mysql/3306/


3. 数据库相关的目录创建出来
配置文件：/etc/my.cnf
/data  是一个单独挂载的一个分区
datadir /home/mysql/mysql3306/data
binlog  /data/mysql/mysql3306/logs
        /data/mysql/mysql3306/tmp
mkdir /home/mysql/3306/{data,logs,tmp} -p


chown -R mysql:mysql 3306/


4. 初始化
cd /usr/local/mysql
/usr/local/mysql/bin/mysqld --defaults-file=/etc/my.cnf --initialize      //需要密码登录，生成data目录下的文件 ，如果没有生成数据库文件 ，会报“pid”错误
/usr/local/mysql/bin/mysqld --defaults-file=/etc/my.cnf --initialize-insecure //不需要密码登录，生成data目录下的文件 ，如果没有生成数据库文件 ，会报“pid”错误
(mysql -h 192.168.23.3 -uroot -P3308 OR mysql)


报错：
./bin/mysqld: error while loading shared libraries: libaio.so.1: cannot open shared object file: No such file or directory
yum install -y libaio  //安装后在初始化就OK了

5. 启动
cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql
/etc/init.d/mysql start

OR 

/usr/local/mysql/bin/mysqld --defaults-file=/etc/my.cnf & (生成.sock文件)

/usr/local/mysql/bin/mysqld --defaults-file=/home/mysql/3306/my.cnf &
